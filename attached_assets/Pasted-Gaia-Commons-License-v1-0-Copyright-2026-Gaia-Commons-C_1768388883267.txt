        // Interactive demo functionality
        function updateDemo() {
            const carbonTax = document.getElementById('carbonTax').value;
            const renewableSubsidy = document.getElementById('renewableSubsidy').value;
            const climateTarget = document.getElementById('climateTarget').value;
            
            // Update display values
            document.getElementById('carbonTaxValue').textContent = `$${carbonTax}`;
            document.getElementById('renewableSubsidyValue').textContent = `$${renewableSubsidy}`;
            document.getElementById('climateTargetValue').textContent = climateTarget;
            
            // Calculate results (simplified Gaia model)
            const policyStrength = (carbonTax / 200) * 0.7 + (renewableSubsidy / 100) * 0.3;
            
            const temperature = 2.5 - (policyStrength * 1.1);
            const score = Math.min(1.0, policyStrength * 0.9 + 0.1);
            const renewables = Math.min(0.95, 0.32 + policyStrength * 0.6);
            const jobs = Math.floor(policyStrength * 10000000);
            const economic = policyStrength * 15.0;
            
            // Update results display
            document.getElementById('resultTemp').textContent = `${temperature.toFixed(1)}¬∞C`;
            document.getElementById('resultScore').textContent = score.toFixed(3);
            document.getElementById('resultRenewables').textContent = `${Math.floor(renewables * 100)}%`;
            document.getElementById('resultJobs').textContent = jobs.toLocaleString();
            document.getElementById('resultEconomic').textContent = `$${economic.toFixed(1)}T`;
            
            // Update colors based on climate target achievement
            const resultsDiv = document.getElementById('demoResults');
            if (temperature <= parseFloat(climateTarget)) {
                resultsDiv.style.background = '#d4edda';
                resultsDiv.style.borderLeft = '4px solid #27ae60';
            } else if (temperature <= 2.0) {
                resultsDiv.style.background = '#fff3cd';
                resultsDiv.style.borderLeft = '4px solid #f39c12';
            } else {
                resultsDiv.style.background = '#f8d7da';
                resultsDiv.style.borderLeft = '4px solid #e74c3c';
            }
        }
        
        function launchPlatform() {
            // Redirect to full platform
            window.open('/platform', '_blank');
        }
        
        function showInstallation() {
            // Show installation modal/section
            alert('Installation:\n\npip install gaia-commons-council\ngaia init my_project\ngaia run scenario.yaml');
        }
        
        // Initialize demo
        updateDemo();
        
        // Smooth scrolling
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                document.querySelector(this.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });

        function showTab(tabName) {
            // Hide all content
            document.querySelectorAll('[id$="-content"]').forEach(el => el.style.display = 'none');
            
            // Remove active class from tabs
            document.querySelectorAll('.tab').forEach(el => el.classList.remove('active'));
            
            // Show selected content and activate tab
            document.getElementById(`${tabName}-content`).style.display = 'block';
            event.target.classList.add('active');
        }
        
        async function runScenario() {
            const button = document.getElementById('runButton');
            button.disabled = true;
            button.textContent = 'üîÑ Running...';
            
            // Build scenario from form
            const scenario = {
                name: document.getElementById('scenarioName').value,
                start_year: 2025,
                end_year: 2050,
                policy: {
                    energy: [
                        {
                            action: "carbon_tax",
                            magnitude: parseInt(document.getElementById('carbonTaxSlider').value),
                            duration: 25
                        },
                        {
                            action: "renewable_subsidy", 
                            magnitude: parseInt(document.getElementById('renewableSubsidySlider').value),
                            duration: 15
                        }
                    ]
                },
                constraints: {
                    climate_limit: parseFloat(document.getElementById('climateTargetSlider').value),
                    equity_weight: parseFloat(document.getElementById('equityWeightSlider').value) / 100
                }
            };
            
            try {
                // Call Gaia API
                const response = await fetch('/api/v1/scenarios/run', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify(scenario)
                });
                
                if (response.ok) {
                    const data = await response.json();
                    currentResults = data.result;
                    displayResults(currentResults);
                    
                    // Show results tab
                    showTab('results');
                } else {
                    throw new Error('Scenario execution failed');
                }
                
            } catch (error) {
                alert(`Error running scenario: ${error.message}`);
            } finally {
                button.disabled = false;
                button.textContent = 'üöÄ Run Transformation';
            }
        }
